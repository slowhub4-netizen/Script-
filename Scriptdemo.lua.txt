-- Fly Toggle
local function startFly()
    local char = player.Character
    if char then
        local root = char:FindFirstChild("HumanoidRootPart")
        local hum = char:FindFirstChild("Humanoid")
        if root and hum then
            hum.PlatformStand = true
            flying_bg = Instance.new("BodyGyro", root)
            flying_bg.MaxTorque = Vector3.new(9e9, 9e9, 9e9)
            flying_bv = Instance.new("BodyVelocity", root)
            flying_bv.MaxForce = Vector3.new(9e9, 9e9, 9e9)
            flying_bv.Velocity = Vector3.new(0, 0.1, 0)
            spawn(function()
                while fly_on and root.Parent do
                    flying_bg.CFrame = camera.CFrame
                    flying_bv.Velocity = Vector3.new(0, 0, 0)
                    wait()
                end
            end)
        end
    end
end
