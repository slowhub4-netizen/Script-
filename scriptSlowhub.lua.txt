-- Minimize logic
local minimized = false
MinimizeButton.MouseButton1Click:Connect(function()
    minimized = not minimized
    ContentFrame.Visible = not minimized
    Sidebar.Visible = not minimized
end)

-- Close confirm
CloseButton.MouseButton1Click:Connect(function()
    local Confirm = Instance.new("Frame")
    Confirm.Size = UDim2.new(0, 250, 0, 120)
    Confirm.Position = UDim2.new(0.5, -125, 0.5, -60)
    Confirm.BackgroundColor3 = Color3.fromRGB(50,50,50)
    Confirm.Parent = ScreenGui

    local Label = Instance.new("TextLabel")
    Label.Size = UDim2.new(1, 0, 0.5, 0)
    Label.Text = "Bạn có muốn thoát script?"
    Label.TextColor3 = Color3.fromRGB(255,255,255)
    Label.Font = Enum.Font.SourceSansBold
    Label.Parent = Confirm

    local Yes = Instance.new("TextButton")
    Yes.Text = "Có"
    Yes.Size = UDim2.new(0.5, 0, 0.5, 0)
    Yes.Position = UDim2.new(0, 0, 0.5, 0)
    Yes.BackgroundColor3 = Color3.fromRGB(100,200,100)
    Yes.Parent = Confirm

    local No = Instance.new("TextButton")
    No.Text = "Không"
    No.Size = UDim2.new(0.5, 0, 0.5, 0)
    No.Position = UDim2.new(0.5, 0, 0.5, 0)
    No.BackgroundColor3 = Color3.fromRGB(200,100,100)
    No.Parent = Confirm

    Yes.MouseButton1Click:Connect(function()
        ScreenGui:Destroy()
    end)

    No.MouseButton1Click:Connect(function()
        Confirm:Destroy()
    end)
end)
