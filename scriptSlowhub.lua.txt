-- 3. H√ÄM T·∫†O TAB V√Ä CHUY·ªÇN TAB
-- ==================================================

local TabPages = {} -- L∆∞u tr·ªØ c√°c Frame n·ªôi dung (page)
local TabIndex = 0

local function SwitchTab(tabName)
    for _, page in pairs(TabPages) do
        page.Visible = false -- ·∫®n t·∫•t c·∫£
    end
    if TabPages[tabName] then
        TabPages[tabName].Visible = true -- Hi·ªán tab ƒë∆∞·ª£c ch·ªçn
    end
end

local function CreateTab(tabName)
    TabIndex = TabIndex + 1
    local positionY = (TabIndex - 1) * 40 + 10 -- TƒÉng 40 pixel cho m·ªói n√∫t

    -- 1. T·∫°o Frame N·ªôi dung (N∆°i ch·ª©a Controls)
    local TabContent = Instance.new("Frame")
    TabContent.Name = tabName .. "Page"
    TabContent.Size = UDim2.new(1, 0, 1, 0)
    TabContent.BackgroundTransparency = 1
    TabContent.Parent = Content
    TabContent.Visible = false
    TabPages[tabName] = TabContent

    -- 2. T·∫°o N√∫t tr√™n thanh Sidebar
    local TabButton = Instance.new("TextButton", Sidebar)
    TabButton.Size = UDim2.new(1, -10, 0, 35)
    TabButton.Position = UDim2.new(0, 5, 0, positionY)
    TabButton.Text = tabName
    TabButton.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    TabButton.TextColor3 = Color3.fromRGB(0, 0, 0)
    TabButton.Font = Enum.Font.SourceSansBold
    TabButton.TextSize = 18
    makeRound(TabButton, 6)

    -- 3. G·∫Øn Logic chuy·ªÉn tab
    TabButton.MouseButton1Click:Connect(function()
        SwitchTab(tabName)
    end)
    
    return TabContent -- Tr·∫£ v·ªÅ Frame n·ªôi dung ƒë·ªÉ b·∫°n c√≥ th·ªÉ th√™m Controls
end
