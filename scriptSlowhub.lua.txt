-- LocalScript đặt trong StarterGui
-- SlowHub UI Demo

-- Tạo ScreenGui
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "SlowHub"
ScreenGui.ResetOnSpawn = false
ScreenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

-- Tạo Frame chính (hub window)
local MainFrame = Instance.new("Frame")
MainFrame.Size = UDim2.new(0, 400, 0, 300)
MainFrame.Position = UDim2.new(0.5, -200, 0.5, -150)
MainFrame.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
MainFrame.BorderSizePixel = 0
MainFrame.Visible = true
MainFrame.Parent = ScreenGui

-- Bo góc
local UICorner = Instance.new("UICorner")
UICorner.CornerRadius = UDim.new(0, 10)
UICorner.Parent = MainFrame

-- Thanh trên (Title Bar)
local TitleBar = Instance.new("Frame")
TitleBar.Size = UDim2.new(1, 0, 0, 40)
TitleBar.BackgroundColor3 = Color3.fromRGB(35, 35, 35)
TitleBar.BorderSizePixel = 0
TitleBar.Parent = MainFrame

local TitleCorner = Instance.new("UICorner")
TitleCorner.CornerRadius = UDim.new(0, 10)
TitleCorner.Parent = TitleBar

-- Tên hub
local Title = Instance.new("TextLabel")
Title.Text = "SlowHub"
Title.Size = UDim2.new(1, -80, 1, 0)
Title.Position = UDim2.new(0, 10, 0, 0)
Title.BackgroundTransparency = 1
Title.TextColor3 = Color3.fromRGB(255, 255, 255)
Title.TextSize = 20
Title.Font = Enum.Font.SourceSansBold
Title.TextXAlignment = Enum.TextXAlignment.Left
Title.Parent = TitleBar

-- Nút minimize/maximize
local ToggleButton = Instance.new("TextButton")
ToggleButton.Text = "-"
ToggleButton.Size = UDim2.new(0, 40, 0, 30)
ToggleButton.Position = UDim2.new(1, -50, 0.5, -15)
ToggleButton.BackgroundColor3 = Color3.fromRGB(255, 80, 80)
ToggleButton.TextColor3 = Color3.fromRGB(255, 255, 255)
ToggleButton.Font = Enum.Font.SourceSansBold
ToggleButton.TextSize = 20
ToggleButton.Parent = TitleBar

local ToggleCorner = Instance.new("UICorner")
ToggleCorner.CornerRadius = UDim.new(0, 8)
ToggleCorner.Parent = ToggleButton

-- Nội dung trong frame
local Content = Instance.new("Frame")
Content.Size = UDim2.new(1, -20, 1, -60)
Content.Position = UDim2.new(0, 10, 0, 50)
Content.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
Content.BorderSizePixel = 0
Content.Parent = MainFrame

local ContentCorner = Instance.new("UICorner")
ContentCorner.CornerRadius = UDim.new(0, 8)
ContentCorner.Parent = Content

-- Label demo
local DemoLabel = Instance.new("TextLabel")
DemoLabel.Size = UDim2.new(1, 0, 0, 50)
DemoLabel.Position = UDim2.new(0, 0, 0, 10)
DemoLabel.BackgroundTransparency = 1
DemoLabel.Text = "Menu demo - thêm chức năng sau"
DemoLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
DemoLabel.TextSize = 18
DemoLabel.Font = Enum.Font.SourceSansItalic
DemoLabel.Parent = Content

-- Biến trạng thái
local minimized = false

-- Script toggle minimize/maximize
ToggleButton.MouseButton1Click:Connect(function()
	if minimized then
		MainFrame:TweenSize(UDim2.new(0, 400, 0, 300), "Out", "Quad", 0.3, true)
		ToggleButton.Text = "-"
	else
		MainFrame:TweenSize(UDim2.new(0, 400, 0, 40), "Out", "Quad", 0.3, true)
		ToggleButton.Text = "+"
	end
	minimized = not minimized
end)
