-- Function th√™m g√≥c bo
local function makeRound(obj, radius)
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, radius)
    corner.Parent = obj
end

-- Function Qu·∫£n l√Ω Tab (T·ªëi ∆∞u h√≥a vi·ªác chuy·ªÉn ƒë·ªïi)
local function switchTab(tabName)
    -- L·∫∑p qua t·∫•t c·∫£ c√°c Khung N·ªôi dung
    for name, frame in pairs(ContentFrames) do
        frame.Visible = (name == tabName) -- Ch·ªâ hi·ªán th·ªã khung n·ªôi dung kh·ªõp v·ªõi tabName
    end

    -- L·∫∑p qua t·∫•t c·∫£ c√°c N√∫t Tab ƒë·ªÉ thay ƒë·ªïi m√†u s·∫Øc
    for name, button in pairs(Tabs) do
        -- ƒê·∫∑t m√†u n·ªÅn kh√°c bi·ªát cho tab ƒëang ho·∫°t ƒë·ªông ƒë·ªÉ l√†m n·ªïi b·∫≠t
        if name == tabName then
            button.BackgroundColor3 = Color3.fromRGB(200, 200, 200) -- M√†u l√†m n·ªïi b·∫≠t (m√†u t·ªëi h∆°n m·ªôt ch√∫t)
        else
            button.BackgroundColor3 = Color3.fromRGB(255, 255, 255) -- M√†u n·ªÅn m·∫∑c ƒë·ªãnh
        end
    end
end

-- Function t·∫°o N√∫t Tab (Sidebar)
local function createTabButton(tabName, yOffset)
    local TabBtn = Instance.new("TextButton", Sidebar)
    TabBtn.Name = tabName
    TabBtn.Size = UDim2.new(1, -20, 0, 40)
    TabBtn.Position = UDim2.new(0, 10, 0, yOffset)
    TabBtn.Text = tabName
    TabBtn.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    TabBtn.TextColor3 = Color3.fromRGB(0, 0, 0)
    TabBtn.Font = Enum.Font.SourceSansBold
    TabBtn.TextSize = 18
    makeRound(TabBtn, 8)

    -- Logic chuy·ªÉn tab khi click
    TabBtn.MouseButton1Click:Connect(function()
        switchTab(tabName)
    end)
    
    Tabs[tabName] = TabBtn
    return TabBtn
end

-- Function t·∫°o Khung N·ªôi dung Tab
local function createContentFrame(tabName)
    local ContentFrame = Instance.new("Frame", ContentArea)
    ContentFrame.Name = tabName .. "Content"
    ContentFrame.Size = UDim2.new(1, 0, 1, 0)
    ContentFrame.BackgroundTransparency = 1 -- L√†m trong su·ªët ƒë·ªÉ ch·ªâ hi·ªán n·ªôi dung b√™n trong
    ContentFrame.Visible = false
    
    ContentFrames[tabName] = ContentFrame
    return ContentFrame
end
