-- @param class T√™n l·ªõp Roblox (v√≠ d·ª•: "Frame", "TextLabel")
-- @param parent ƒê·ªëi t∆∞·ª£ng cha
-- @param properties B·∫£ng ch·ª©a c√°c thu·ªôc t√≠nh v√† gi√° tr·ªã
local function Create(class, parent, properties)
    local obj = Instance.new(class)
    for prop, value in pairs(properties) do
        obj[prop] = value
    end
    obj.Parent = parent
    return obj
end

--- Th√™m m·ªôt UICorner cho ƒë·ªëi t∆∞·ª£ng ƒë·ªÉ l√†m tr√≤n g√≥c.
-- @param obj ƒê·ªëi t∆∞·ª£ng UI ƒë·ªÉ l√†m tr√≤n
-- @param radius B√°n k√≠nh l√†m tr√≤n (s·ªë nguy√™n)

    -- T·∫°o v√† thi·∫øt l·∫≠p UICorner
    Create("UICorner", obj, {

--- T√≠nh to√°n v·ªã tr√≠ Y ti·∫øp theo cho m·ªôt Control trong ScrollingFrame v√† c·∫≠p nh·∫≠t CanvasSize.
-- H·ªó tr·ª£ t·ª± ƒë·ªông x·∫øp ch·ªìng c√°c ƒëi·ªÅu khi·ªÉn.
-- @param ParentScrollFrame ScrollingFrame ch·ª©a c√°c controls
-- @return V·ªã tr√≠ Y offset m·ªõi cho control ti·∫øp theo
local function GetNextControlY(ParentScrollFrame)
    -- L·∫•y v·ªã tr√≠ Y hi·ªán t·∫°i c·ªßa n·ªôi dung
    local currentY = ParentScrollFrame.CanvasSize.Y.Offset
    
    -- T√≠nh to√°n v·ªã tr√≠ Y m·ªõi cho control (v·ªã tr√≠ hi·ªán t·∫°i + padding)
    local newY = currentY + Configs.CONTROL_PADDING
    
    -- C·∫≠p nh·∫≠t CanvasSize ƒë·ªÉ ScrollFrame c√≥ th·ªÉ cu·ªôn ƒë·∫øn v·ªã tr√≠ m·ªõi
    local newCanvasY = newY + Configs.CONTROL_HEIGHT + Configs.CONTROL_PADDING
    ParentScrollFrame.CanvasSize = UDim2.new(0, 0, 0, newCanvasY)
    
    return newY
end

--- T·∫°o m·ªôt Tab m·ªõi (N√∫t Tab v√† Khung N·ªôi dung Tab).
-- @param tabName T√™n Tab hi·ªÉn th·ªã tr√™n n√∫t v√† ƒë∆∞·ª£c d√πng l√†m kh√≥a (key)
-- @return ScrollingFrame n∆°i c√°c Controls s·∫Ω ƒë∆∞·ª£c th√™m v√†o
function MakeTab(tabName)
    TabIndex = TabIndex + 1
    -- T√≠nh to√°n v·ªã tr√≠ Y cho n√∫t tab trong Sidebar
    local positionY = (TabIndex - 1) * (Configs.CONTROL_HEIGHT + 5) + Configs.CONTROL_PADDING 

    -- 1. T·∫°o Khung N·ªôi dung Tab (Tab Content Frame - Page)
    local TabContent = Create("Frame", HubControls.Content, {
        Size = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        Visible = false, -- M·∫∑c ƒë·ªãnh ·∫©n, ch·ªâ hi·ªÉn th·ªã tab ƒëang ƒë∆∞·ª£c ch·ªçn
    })

    -- 2. T·∫°o ScrollingFrame b√™n trong Tab Content
    -- ƒê√¢y l√† n∆°i c√°c Controls s·∫Ω ƒë∆∞·ª£c th√™m v√†o, cho ph√©p cu·ªôn khi n·ªôi dung d√†i
    local ScrollFrame = Create("ScrollingFrame", TabContent, {
        Size = UDim2.new(1, 0, 1, 0),
        BackgroundTransparency = 1,
        CanvasSize = UDim2.new(0, 0, 0, 0), -- CanvasSize ƒë∆∞·ª£c c·∫≠p nh·∫≠t t·ª± ƒë·ªông b·ªüi c√°c Controls
        ScrollBarThickness = 6,
        ScrollBarImageColor3 = Configs.ACCENT_COLOR,
        ScrollBarBackgroundTransparency = 0.8
    })
    
    -- L∆∞u tr·ªØ tham chi·∫øu n·ªôi dung tab
    TabPages[tabName] = {Content = TabContent, Scroll = ScrollFrame}

    -- 3. T·∫°o N√∫t Tab trong Sidebar
    local TabButton = Create("TextButton", HubControls.Sidebar, {
        Size = UDim2.new(1, -20, 0, Configs.CONTROL_HEIGHT),
        Position = UDim2.new(0, 10, 0, positionY),
        Text = tabName,
        BackgroundColor3 = Configs.SIDEBAR_COLOR, -- M√†u m·∫∑c ƒë·ªãnh c·ªßa sidebar
        TextColor3 = Configs.TEXT_COLOR,
        Font = Enum.Font.SourceSansBold,
        TextSize = 16,
    })
    AddCorner(TabButton, 6)

    -- 4. Logic chuy·ªÉn ƒë·ªïi Tab (Tab switching logic)
    local function switchTab()
        -- ·∫®n t·∫•t c·∫£ n·ªôi dung tab
        for name, page in pairs(TabPages) do
            page.Content.Visible = false
            -- ƒê·∫∑t l·∫°i m√†u n·ªÅn c·ªßa t·∫•t c·∫£ c√°c n√∫t tab
            Tabs[name].BackgroundColor3 = Configs.SIDEBAR_COLOR

        -- Hi·ªÉn th·ªã n·ªôi dung tab hi·ªán t·∫°i v√† l√†m n·ªïi b·∫≠t n√∫t tab
        TabContent.Visible = true
        TabButton.BackgroundColor3 = Configs.ACCENT_COLOR 
    end

    -- K·∫øt n·ªëi s·ª± ki·ªán nh·∫•p chu·ªôt
    TabButton.MouseButton1Click:Connect(switchTab)
    Tabs[tabName] = TabButton

    -- Tr·∫£ v·ªÅ ScrollFrame ƒë·ªÉ th√™m c√°c controls sau n√†y
    return ScrollFrame 
end
